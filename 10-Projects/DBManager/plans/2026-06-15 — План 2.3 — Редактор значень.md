> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`).

**Goal:** Додавати, редагувати й видаляти значення з адмінки, обираючи **область дії**: на **групу** (одна зміна → усі сайти групи) чи на **сайт** (для одного сайта / перекриття групового значення). Після збереження — публікація уражених сайтів у DataBridge. Це закриває «як додати/змінити дані для одного сайта» і «як змінити для кількох сайтів одночасно» (групове значення).

**Architecture:** Новий Livewire-компонент `ValueEditor` (модальна форма / бічна панель) поряд із `ValuesGrid`. Грід має кнопку «+ Додати значення» і відкриває редактор на не-телефонних рядках (телефонні відкривають `SlotPanel` із 2.2). Редактор створює/оновлює `DataValue` з обраним `scope_type`+`scope_id`; кнопка «Перекрити для цього сайта» на груповому рядку робить site-scoped копію. Уражені сайти = scope_type `site` → один сайт; `group` → усі сайти групи. Публікація — `SitePayloadCompiler::publish` + `BridgePublisher::push` (як 2.2 Save). Аудит — запис `AuditLog` на create/update/delete.

**Tech Stack:** Laravel, Livewire 3, Tailwind, PHPUnit + Livewire::test, Http::fake.

**Спека:** `DBManager — Дизайн.md` §4 (область дії: група/сайт-перекриття, спільні значення без дублів), §7 (грід, дії). Макет: `grid-failover-v5.html` (бейджі «Група» / «✎ цього сайта»).

**Моделі/сервіси:** `App\Models\DataValue` (key, value_type_id, scope_type ['group'|'site'], scope_id, content[array cast], status['active']; фабрика `forSite(Site)`/`forGroup(SiteGroup)`/`ofType(code)`). `App\Models\ValueType` (code, name; `firstOrCreate(['code'=>...],['name'=>...])`). `App\Models\Site` (group, site_group_id). `App\Models\SiteGroup`. `App\Models\AuditLog::create([actor_type,action,subject_type,subject_id,old,new])`. `App\Services\Publishing\SitePayloadCompiler::publish(Site)`, `App\Services\Publishing\BridgePublisher::push(Publication)`.

```php
<?php

namespace Tests\Feature\Admin;

use App\Admin\AffectedSites;
use App\Models\DataValue;
use App\Models\Site;
use App\Models\SiteGroup;
use Illuminate\Foundation\Testing\RefreshDatabase;
use Tests\TestCase;

class AffectedSitesTest extends TestCase
{
    use RefreshDatabase;

    public function test_site_scope_returns_single_site(): void
    {
        $site = Site::factory()->create();
        $value = DataValue::factory()->forSite($site)->create();

        $sites = app(AffectedSites::class)->for($value);

        $this->assertSame([$site->id], $sites->pluck('id')->all());
    }

    public function test_group_scope_returns_all_group_sites(): void
    {
        $group = SiteGroup::factory()->create();
        $a = Site::factory()->for($group, 'group')->create();
        $b = Site::factory()->for($group, 'group')->create();
        $value = DataValue::factory()->forGroup($group)->create();

        $ids = app(AffectedSites::class)->for($value)->pluck('id')->sort()->values()->all();

        $this->assertSame([$a->id, $b->id], $ids);
    }
}
```

<?php

namespace App\Admin;

use App\Models\DataValue;
use App\Models\Site;
use Illuminate\Support\Collection;

class AffectedSites
{
    /** @return Collection<int, Site> сайти, на які впливає значення */
    public function for(DataValue $value): Collection
    {
        return $value->scope_type === 'site'
            ? Site::where('id', $value->scope_id)->get()
            : Site::where('site_group_id', $value->scope_id)->get();
    }
}

public function test_create_group_value(): void
{
    $this->actingAs(\App\Models\User::factory()->create());
    $group = \App\Models\SiteGroup::factory()->create();
    $site = \App\Models\Site::factory()->for($group,'group')->create();

    \Livewire\Livewire::test(\App\Livewire\ValueEditor::class)
        ->call('createFor', $site->id)
        ->set('type','price')->set('key','price_basic')->set('value','1200')->set('scope','group')
        ->call('save');

    $dv = \App\Models\DataValue::where('key','price_basic')->first();
    $this->assertNotNull($dv);
    $this->assertSame('group', $dv->scope_type);
    $this->assertSame($group->id, $dv->scope_id);
    $this->assertSame('1200', $dv->content['value']);
    $this->assertTrue(\App\Models\AuditLog::where('action','value.created')->exists());
}

public function test_create_site_value(): void
{
    $this->actingAs(\App\Models\User::factory()->create());
    $site = \App\Models\Site::factory()->create();
    \Livewire\Livewire::test(\App\Livewire\ValueEditor::class)
        ->call('createFor', $site->id)
        ->set('type','text')->set('key','addr')->set('value','вул. Хрещатик, 1')->set('scope','site')
        ->call('save');
    $dv = \App\Models\DataValue::where('key','addr')->first();
    $this->assertSame('site', $dv->scope_type);
    $this->assertSame($site->id, $dv->scope_id);
}

public function test_validation_requires_key_and_type(): void

        ->call('createFor', $site->id)

}
